def replace_method(text: str, signature: str, new_method: str) -> str:
    start = text.find(signature)
    if start < 0:
        raise RuntimeError(f'Nie znaleziono: {signature}')
    brace = text.find('{', start)
    if brace < 0:
        raise RuntimeError(f'Brak klamry: {signature}')
    depth = 0
    in_string = False
    in_char = False
    escaped = False
    i = brace
    while i < len(text):
        ch = text[i]
        if in_string:
            if escaped:
                escaped = False
            elif ch == '\\':
                escaped = True
            elif ch == '"':
                in_string = False
        elif in_char:
            if escaped:
                escaped = False
            elif ch == '\\':
                escaped = True
            elif ch == "'":
                in_char = False
        else:
            if ch == '"':
                in_string = True
            elif ch == "'":
                in_char = True
            elif ch == '{':
                depth += 1
            elif ch == '}':
                depth -= 1
                if depth == 0:
                    return text[:start] + new_method.rstrip() + text[i + 1:]
        i += 1
    raise RuntimeError(f'Nie domknięto: {signature}')
