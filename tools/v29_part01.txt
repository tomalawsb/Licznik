 lowPassSensor(event.values, magneticValues);
            magneticReady = true;
        }

        if (!orientationReady && gravityReady && magneticReady) {
            orientationReady = SensorManager.getRotationMatrix(rotationMatrix, null, gravityValues, magneticValues);
        }
        if (!orientationReady) return;

        float[] orientation = new float[3];
        SensorManager.getOrientation(rotationMatrix, orientation);
        float azimuth = (float) Math.toDegrees(orientation[0]);
        if (azimuth < 0f) azimuth += 360f;
        rotateCompassTo(-azimuth);
    }

    private void lowPassSensor(float[] input, float[] output) {
        final float alpha = 0.18f;
        for (int i = 0; i < 3 && i < input.length; i++) {
            output[i] += alpha * (input[i] - output[i]);
        }
    }

    private void rotateCompassTo(float targetRotation) {
        float delta = ((targetRotation - compassDisplayRotation + 540f) % 360f) - 180f;
        compassDisplayRotation += delta * 0.32f;
        if (compassView != null) compassView.setRotation(compassDisplayRotation);
    }

    @Override public void onAccuracyChanged(Sensor sensor, int accuracy) {
        // Nie wymaga dodatkowej obsługi.
    }
